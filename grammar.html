Grammar of oak

Expression

Expression:
	IfElseExpression
IfElseExpression:
	AssignExpression
	if Expression : Expression else IfElseExpression
AssignExpression:
	PipelineExpression
	PipelineExpression   = AssignExpression
	PipelineExpression  += AssignExpression
	PipelineExpression  -= AssignExpression
	PipelineExpression ++= AssignExpression
	PipelineExpression  *= AssignExpression
	PipelineExpression  /= AssignExpression
	PipelineExpression  %= AssignExpression
	PipelineExpression ^^= AssignExpression
	PipelineExpression  &= AssignExpression
	PipelineExpression  ^= AssignExpression
	PipelineExpression  |= AssignExpression
PipelineExpression:
	AppExpression
	PipelineExpression => AppExpression
AppExpression:
	OrExpression
	OrExpression app AppExpression
OrExpression:
	XorExpression
	OrExpression || XorExpression
XorExpression:
	AndExpression
	XorExpression ^^ AndExpression
AndExpression:
	BitOrExpression
	AndExpression && BitOrExpression
BitOrExpression:
	BitXorExpression
	BitOrExpression | BitXorExpression
BitXorExpression:
	BitAndExpression
	BitXorExpression ^ BitAndExpression
BitAndExpression:
	CmpExpression
	BitAndExpression & CmpExpression
CmpExpression:
	ShiftExpression
	ShiftExpression   < ShiftExpression
	ShiftExpression  <= ShiftExpression
	ShiftExpression   > ShiftExpression
	ShiftExpression  >= ShiftExpression
	ShiftExpression  == ShiftExpression
	ShiftExpression  != ShiftExpression
	ShiftExpression  in ShiftExpression
	ShiftExpression !in ShiftExpression
	ShiftExpression  is ShiftExpression
	ShiftExpression !is ShiftExpression
ShiftExpression:
	AddExpression
	ShiftExpression  << AddExpression
	ShiftExpression  >> AddExpression
	ShiftExpression >>> AddExpression
AddExpression:
	MulExpression
	AddExpression  + MulExpression
	AddExpression  - MulExpression
	AddExpression ++ MulExpression
MulExpression:
	MulExpression
	MulExpression * PowExpression
	MulExpression / PowExpression
	MulExpression % PowExpression
PowExpression:
	ApplyExpression
	ApplyExpression ** PowExpression
ApplyExpression:
	UnaryExpression
	ApplyExpression UnaryExpression
UnaryExpression:
	IndexingExpression
	-- UnaryExpression
	 ~ UnaryExpression
	 # UnaryExpression
	 ! UnaryExpression
IndexingExpression:
	CompositionExpression
	IndexingExpression ![ Expressions ]
	IndexingExpression ![ Slicings ]
CompositionExpression:
	AscribeExpression
	CompositionExpression @ PowExpression
AscribeExpression:
	DotExpression
	AscibeExpression as Type
DotExpression:
	PrimaryExpression
	DotExpression . identifier
	DotExpression . TemplateInstance
	DotExpression . NewExpression
PrimaryExpression:
	integer
	real_number
	string_literal
	true
	false
	null
	this
	super
	$
	_.opt identifier
	_.opt TemplateInstance
	_
	NewExpression
	LambdaExpression
	TupleExpression
	ArrayLiteral
	AssocArrayLiteral
	StructLiteral
	BasicType . identifier
	Typeof . identifier
	Typeid
	Mixin
	mixin ( Expression )
NewExpression:
	new Type
LambdaExpression:
	\ : Type opt FunctionArgumentsopt BlockStatement
	\ : Type opt FunctionArgumentsopt = PrimaryExpression
TupleExpression:
	( )
	( Expressions ,opt )

Type

Type:
	FunctionType
FunctionType:
	PointerType
	PointerType -> FunctionType
PointerType:
	PrimaryType
	# PointerType
PrimaryType:
	BuiltInType
	ArrayType
	AssocArrayType
	SymbolType
	TupleType
	Typeof
	Mixin
	StructType
BuiltInType:
	int32
	uint32
	int64
	uint64
	real32
	real64
	bool
	unit
	string
	char
ArrayType:
	[ Type ]
	
AssocArrayType:
	[ Type : Type ]
SymbolType:
	_.opt identifier
	_.opt TemplateInstance
	SymbolType . identifier
	SymbolType . TemplateInstance
TupleType:
	( Types ,opt )
StructType:
	struct { StructBodyList }
	
StructBodyList:
	StructBody
	StructBody StructBodyList

StructBody:
	Declaration

Statment

Statment:
	Declaration
	ExpressionStatement
	LabelStatement
	WhileStatement
	DoWhileStatement
	ForStatement
	ForaechStatement
	ForeachReverseStatement
	BreakStatement
	ContinueStatement
	GotoStatement
	ReturnStatement
	BlockStatement
	MixinStatement
ExpressionStatement:
	Expression ;
LabelStatement:
	identifier :
WhileStatement:
	while Expression : Statement
DoWhileStatement:
	do Statement while Expression ;
ForStatement:
	for Statement Expression ; Expression : Statement
ForeachStatement:
	foreach         ForeachArguments , opt ; Expression : Statement
	foreach         ForeachArgument , opt  ; Expression .. Expression : Statement

ForeachReverseStatement:
	foreach_reverse ForeachArguments , opt ; Expression : Statement
	foreach_reverse ForeachArgument , opt  ; Expression .. Expression : Statement

ForeachArguments:
	ForeachArgument
	ForeachArgument , ForeachArguments

ForeachArgument:
	identifier : Type opt
BreakStatement:
	break identifieropt ;
ContinueStatement:
	continue identifieropt ;
GotoStatement:
    goto identifier ;
ReturnStatement:
	return Expressionopt ;
BlockStatement:
	{ StatementList }

StatementList:
	Statement StatementList
	Statement
MixinStatement:
	Mixin ;

Declaration

Declartion:
	LetDeclaration
	FunctionDeclaration
	TypedefDeclaration
LetDeclartion:
	let LetDeclBodies ,opt ;

LetDeclBodies:
	LetDeclBody
	LetDeclBody , LetDeclBodies

LetDeclBody:
	identifier : Type
	identifier = Expression
	identifier : Type = Expression
FuncDeclartion:
	func identifier : Type opt FunctionArgumentListopt BlockStatement
	func identifier : Type opt FunctionArgumentListopt = Expression ;

FunctionArgumentList:
	FunctionArgument FunctionArgumentList
	FunctionArgument

FunctionArgument:
	identifier : Type opt
	_          : Type opt
TypedefDeclaration:
	typedef identifier =opt Type
